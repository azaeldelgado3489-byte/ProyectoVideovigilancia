from fastapi import APIRouter, UploadFile, File, Form, HTTPException
import os
import psycopg2
from psycopg2.extras import RealDictCursor
import subprocess

router = APIRouter()

ROSTROS_DIR = "/var/cctv/rostros"
CLIPS_DIR = "/var/cctv/clips"

os.makedirs(ROSTROS_DIR, exist_ok=True)
os.makedirs(CLIPS_DIR, exist_ok=True)

TOKEN_VALIDO = "clavesecreta123"

DB_CFG = dict(host="localhost", dbname="cctv_db", user="cctv_user", password="admin")

def db():
    return psycopg2.connect(cursor_factory=RealDictCursor, **DB_CFG)

@router.post("/api/eventos")
async def crear_evento(
    nombre: str = Form(...),
    confianza: float = Form(...),
    tipo: str = Form(...),
    fecha: str = Form(...),
    token: str = Form(...),
    imagen: UploadFile = File(...),
    video: UploadFile = File(...)
):
    if token != TOKEN_VALIDO:
        raise HTTPException(status_code=401, detail="token inválido")

    if imagen is None or not imagen.filename:
        raise HTTPException(status_code=400, detail="no se recibió archivo de imagen")

    if video is None or not video.filename:
        raise HTTPException(status_code=400, detail="no se recibió archivo de video")

    # Guardar imagen
    img_path = os.path.join(ROSTROS_DIR, imagen.filename)
    with open(img_path, "wb") as f:
        f.write(await imagen.read())

    # Guardar archivo temporal
    temp_path = os.path.join(CLIPS_DIR, "temp_" + video.filename)
    with open(temp_path, "wb") as f:
        f.write(await video.read())

    # Ruta final
    vid_path = os.path.join(CLIPS_DIR, video.filename)

    # Si es video, convertir a H.264/AAC
    if video.content_type and video.content_type.startswith("video/"):
        try:
            subprocess.run([
                "ffmpeg", "-y", "-i", temp_path,
                "-c:v", "libx264", "-c:a", "aac",
                vid_path
            ], check=True)
            os.remove(temp_path)
        except Exception as e:
            raise HTTPException(status_code=500, detail=f"fallo al convertir video: {e}")
    else:
        # Si no es video, guardar tal cual
        os.rename(temp_path, vid_path)

    # Insertar en la base de datos
    try:
        ruta_video = video.filename if video and video.filename else ""
        conn = db(); cur = conn.cursor()
        cur.execute("""
            INSERT INTO eventos (
                persona_id,
                nombre_detectado,
                confianza,
                tipo,
                fecha,
                ruta_imagen,
                ruta_video
            ) VALUES (%s, %s, %s, %s, %s, %s, %s);
        """, (
            None,
            nombre,
            confianza,
            tipo,
            fecha,
            imagen.filename,
            ruta_video
        ))
        conn.commit()
        cur.close(); conn.close()
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"fallo al insertar en BD: {e}")

    return {
        "status": "ok",
        "nombre": nombre,
        "confianza": confianza,
        "tipo": tipo,
        "fecha": fecha,
        "imagen": imagen.filename,
        "video": ruta_video
    }

from jinja2 import Environment, FileSystemLoader
from psycopg2.extras import RealDictCursor
import psycopg2
from collections import Counter
from .correo import enviar_correo
import os

# Configuración de la BD
DB_CFG = dict(host="localhost", dbname="cctv_db", user="cctv_user", password="admin")

# Ruta absoluta a la carpeta de plantillas
BASE_DIR = os.path.dirname(os.path.abspath(__file__))
TEMPLATE_DIR = os.path.join(BASE_DIR, '..', 'templates')
env = Environment(loader=FileSystemLoader(TEMPLATE_DIR))

def obtener_eventos_por_dia(dia: str):
    """Consulta los eventos de la BD para un día específico."""
    conn = psycopg2.connect(cursor_factory=RealDictCursor, **DB_CFG)
    cur = conn.cursor()
    cur.execute("""
        SELECT *
        FROM eventos
        WHERE to_char(fecha,'YYYY-MM-DD')=%s
        ORDER BY fecha DESC;
    """, (dia,))
    rows = cur.fetchall()
    cur.close()
    conn.close()
    return rows

def generar_reporte(dia: str, destinatario: str):
    """Genera un reporte HTML con los eventos del día y lo envía por correo."""
    eventos = obtener_eventos_por_dia(dia)

    # Calcular total y hora más concurrida
    total_eventos = len(eventos)
    horas = [e["fecha"].strftime("%H:%M") for e in eventos] if eventos else []
    hora_peak = Counter(horas).most_common(1)[0][0] if horas else "N/A"

    # Renderizar plantilla
    template = env.get_template("reporte.html")
    html_body = template.render(
        dia=dia,
        eventos=eventos,
        total_eventos=total_eventos,
        hora_peak=hora_peak
    )

    # Enviar correo
    enviar_correo(destinatario, f"Reporte de accesos {dia}", html_body)

@app.get("/", response_class=HTMLResponse)
def root(request: Request):
    auth = request.cookies.get("auth") == "1"
    if not auth:
        return RedirectResponse(url="/login", status_code=302)

    hoy = datetime.date.today().strftime("%Y-%m-%d")
    return RedirectResponse(url=f"/galeria/{hoy}", status_code=302)

@app.get("/galeria/{dia}", response_class=HTMLResponse)
def galeria(request: Request, dia: str):
    conn = db(); cur = conn.cursor()

    cur.execute("""
        SELECT DISTINCT to_char(fecha,'YYYY-MM-DD') AS dia
        FROM eventos
        ORDER BY dia DESC;
    """)
    rows = cur.fetchall()
    dias = [r["dia"] for r in rows] if rows else []

    cur.execute("""
        SELECT *
        FROM eventos
        WHERE to_char(fecha,'YYYY-MM-DD')=%s
        ORDER BY fecha DESC;
    """, (dia,))
    eventos_rows = cur.fetchall()

    cur.close(); conn.close()
    hoy = datetime.date.today().strftime("%Y-%m-%d")

    auth = request.cookies.get("auth") == "1"

    return templates.TemplateResponse(
        "galeria.html",
        {
            "request": request,
            "dia": dia,
            "eventos": eventos_rows,
            "dias": dias,
            "hoy": hoy,
            "user_is_authenticated": auth
        }
    )
